page-schema login (
	datiLogin: form(
			Username: text;
			Password: password;
			Login   : submit();
		  );
)
DB to page schema login 
parameter(Username)
(
	if( SELECT login 
	    FROM passeggero 
	    WHERE login=?Username?;) 
	then *bigliettiPage else *login end;
)
DB from page schema login 
(
	form: if( SELECT login 
		  FROM passeggero 
		  WHERE login=?Username? AND password=?Password?;) 
	      then *bigliettiPage else *login end; 		
)
